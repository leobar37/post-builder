Comunicación con OpenCode vía ACP

Agent Client Protocol (ACP) permite comunicación bidireccional con OpenCode mediante stdin/stdout.

OpenCodeBridge - Clase Principal

import { AcpClient } from '@agentclientprotocol/sdk';
import { EventEmitter } from 'events';
import { spawn } from 'child_process';

export class OpenCodeBridge extends EventEmitter {
  private clients: Map = new Map();
  private processes: Map = new Map();

  async createSession(sessionId: string, cwd: string): Promise {
    // Spawn opencode en modo ACP
    const process = spawn('opencode', ['acp'], {
      cwd,
      env: { OPENCODE_API_KEY: process.env.OPENCODE_API_KEY }
    });

    this.processes.set(sessionId, process);

    // Crear cliente ACP
    const client = new AcpClient({
      stdin: process.stdin,
      stdout: process.stdout,
    });

    this.clients.set(sessionId, client);

    // Escuchar mensajes del agente
    client.onMessage((message) => {
      this.handleMessage(sessionId, message);
    });

    // Initialize session
    await client.initialize({
      name: 'video-pipeline-agent',
      version: '1.0.0',
    });
  }

  private handleMessage(sessionId: string, message: Message): void {
    if (message.role === 'assistant' && message.content) {
      this.emit('event', {
        type: 'text',
        sessionId,
        data: { role: 'assistant', content: message.content }
      });
    }

    if (message.tool_calls) {
      this.emit('event', {
        type: 'tool_call',
        sessionId,
        data: message.tool_calls[0]
      });
    }
  }
}

Selección de Modelos por Tarea

// Configuración de modelos
interface OpenCodeModelsConfig {
  default: ModelConfig;
  idea_generation: ModelConfig;  // Claude Opus - Creatividad
  code_generation: ModelConfig;  // Claude Sonnet - Balance
  editing: ModelConfig;          // Claude Haiku - Velocidad
  economy?: ModelConfig;         // GPT-4o-mini - Costo
}

// Uso por tipo de tarea
async sendPrompt(
  sessionId: string,
  prompt: string,
  options: { taskType?: 'idea' | 'code' | 'editing' | 'economy' }
): Promise {
  const modelConfig = this.selectModel(options.taskType);

  // Notificar modelo seleccionado
  this.emit('event', {
    type: 'model_selected',
    data: {
      provider: modelConfig.provider,
      model: modelConfig.model,
      taskType: options.taskType
    }
  });

  await client.sendMessage({
    role: 'user',
    content: prompt,
    model_params: {
      model: modelConfig.model,
      temperature: modelConfig.temperature,
      max_tokens: modelConfig.max_tokens,
      system_prompt: modelConfig.system_prompt
    }
  });
}

React Hook - useOpenCode

export function useOpenCode(options: UseOpenCodeOptions = {}) {
  const [isConnected, setIsConnected] = useState(false);
  const [sessionId, setSessionId] = useState(null);
  const eventSourceRef = useRef(null);

  const sendCommand = useCallback(async ({
    prompt,
    contextPath,
    taskType
  }: {
    prompt: string;
    contextPath?: string;
    taskType?: 'idea' | 'code' | 'editing';
  }) => {
    const response = await fetch('/api/command', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ prompt, contextPath, taskType })
    });

    const result = await response.json();
    setSessionId(result.sessionId);
    connectSSE(result.sessionId);
  }, []);

  const connectSSE = useCallback((sid: string) => {
    const es = new EventSource(`/events/${sid}`);
    eventSourceRef.current = es;

    es.onmessage = (event) => {
      const data = JSON.parse(event.data);

      switch (data.type) {
        case 'text':
          options.onText?.(data.data.content);
          break;
        case 'tool_call':
          options.onToolCall?.(data.data);
          break;
        case 'model_selected':
          console.log('Using model:', data.data.model);
          break;
      }
    };
  }, []);

  return { sendCommand, isConnected, sessionId };
}

API Routes - Express

// POST /api/command - Iniciar sesión
router.post('/', async (req, res) => {
  const { prompt, contextPath, taskType } = req.body;
  const sessionId = uuidv4();

  await bridge.createSession(sessionId, contextPath || process.cwd());

  await bridge.sendPrompt(sessionId, prompt, {
    taskType,  // 'idea' | 'code' | 'editing'
    context: { projectPath: contextPath }
  });

  res.json({
    sessionId,
    status: 'started',
    message: 'Command sent to OpenCode. Listen to SSE for events.'
  });
});

// GET /events/:sessionId - SSE Stream
router.get('/:sessionId', (req, res) => {
  res.setHeader('Content-Type', 'text/event-stream');
  res.setHeader('Cache-Control', 'no-cache');
  res.setHeader('Connection', 'keep-alive');

  const onEvent = (event: OpenCodeEvent) => {
    res.write(`data: ${JSON.stringify(event)}\n\n`);
  };

  bridge.on('event', onEvent);

  req.on('close', () => {
    bridge.off('event', onEvent);
  });
});

Remotion Hot Reload 

// Hook para recarga automática
export function useRemotionReload({
  videoId,
  apiUrl = 'http://localhost:3000',
  onCodeUpdate
}: UseRemotionReloadOptions) {
  const [reloadKey, setReloadKey] = useState(0);

  useEffect(() => {
    const es = new EventSource(`${apiUrl}/api/videos/${videoId}/events`);

    es.onmessage = (event) => {
      const data = JSON.parse(event.data);

      if (data.type === 'code_updated') {
        setReloadKey(k => k + 1);  // Force re-mount
        onCodeUpdate?.();
      }
    };

    return () => es.close();
  }, [videoId]);

  return { reloadKey, forceReload: () => setReloadKey(k => k + 1) };
}
